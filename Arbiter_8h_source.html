100 static int ClipSegmentToLine(ClipVertex vOut[2], ClipVertex vIn[2],
101  const Vec2 &normal, float offset, char clipEdge)
102 {
103  // Start with no output points
104  int numOut = 0;
105 
106  // Calculate the distance of end points to the line
107  float distance0 = Dot(normal, vIn[0].v) - offset;
108  float distance1 = Dot(normal, vIn[1].v) - offset;
109 
110  // If the points are behind the plane
111  if (distance0 <= 0.0f) vOut[numOut++] = vIn[0];
112  if (distance1 <= 0.0f) vOut[numOut++] = vIn[1];
113 
114  // If the points are on different sides of the plane
115  // 若兩點的位置在面的不同邊，會算出其相交的頂點
116  if (distance0 * distance1 < 0.0f)
117  {
118  // Find intersection point of edge and plane
119  float interp = distance0 / (distance0 - distance1);
120  vOut[numOut].v = vIn[0].v + interp * (vIn[1].v - vIn[0].v);
121  if (distance0 > 0.0f)
122  {
123  vOut[numOut].fp = vIn[0].fp;
124  vOut[numOut].fp.e.inEdge1 = clipEdge;
125  vOut[numOut].fp.e.inEdge2 = NO_EDGE;
126  }
127  else
128  {
129  vOut[numOut].fp = vIn[1].fp;
130  vOut[numOut].fp.e.outEdge1 = clipEdge;
131  vOut[numOut].fp.e.outEdge2 = NO_EDGE;
132  }
133  ++numOut;
134  }
